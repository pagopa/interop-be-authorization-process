# How to generate RSA key pair
openssl genrsa -out keypair.rsa.pem 2048
openssl rsa -in keypair.rsa.pem -pubout -out key.rsa.pub
openssl pkcs8 -topk8 -inform PEM -outform PEM -nocrypt -in keypair.rsa.pem -out key.rsa.priv

# How to generate EC key pair
openssl ecparam -genkey -name prime256v1 -noout -out private.ec.key
openssl pkcs8 -topk8 -inform pem -in private.ec.key -outform pem -nocrypt -out private.ec.pem
openssl ec -in private.ec.pem -pubout -out public.ec.pem
